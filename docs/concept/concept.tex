% ----------------------------- PROBLEM -----------------------------
\begin{frame}{Enterprises ship agents they cannot trust}{The problem}
\vspace{0.15cm}
\begin{columns}[T]
  \column{0.60\textwidth}
  \begin{itemize}\setlength\itemsep{8pt}
    \item \textbf{Non-deterministic.} The same instruction yields different tool calls and outputs each run.
    \item \textbf{Failures are silent.} It finishes its calls --- yet mis-triages a ticket or overwrites a page.
    \item \textbf{Exact-match tests fail.} They assume one correct output; agent trajectories have many.
    \item \textbf{Failures don't reproduce.} Re-running changes the output and re-fires side effects.
  \end{itemize}
  \column{0.38\textwidth}
  \statcard{crose}{12/80}{tickets mis-routed, \emph{zero} alerts}\hfill
  \statcard{camber}{0}{reproducible replays from a log}\\[7pt]
  \statcard{cindigo}{?}{which step actually failed}\hfill
  \statcard{cblue}{3 wk}{later: ``what did it see?''}
\end{columns}
\vspace{0.3cm}
\begin{center}\footnotesize\color{slate}
No continuous evaluation \dt\ no deterministic debugging \dt\ no audit trail for production AI agents.
\end{center}
\end{frame}

% ----------------------------- CORE AI MECHANISM -----------------------------
\begin{frame}{AI is the mechanism, not a feature}{Core AI mechanism}
{\footnotesize\color{slate}Remove the models and nothing works --- every output is reasoning over live signals, not keyword search.}
\vspace{0.12cm}
\begin{columns}[T]
  \column{0.49\textwidth}
  \begin{itemize}\setlength\itemsep{3pt}
    \item \textbf{Semantic retrieval} --- BGE-768 finds similar incidents and runbooks, with attribution.
    \item \textbf{Structured generation} --- Llama-3.1-8B drafts the RCA and duplicate verdict as typed objects.
    \item \textbf{LLM-as-judge} --- scores 4 rubric dimensions, run twice with order reversed to calibrate bias.
  \end{itemize}
  \column{0.49\textwidth}
  \begin{itemize}\setlength\itemsep{3pt}
    \item \textbf{Safety classifier} --- Llama-Guard-3 pre-filters every transcript.
    \item \textbf{Failure attribution} --- a DAG blames retrieval, planning, or execution.
    \item \textbf{Evaluating the evaluator} --- Cohen's $\kappa$ vs.\ a gold set; centroid \textbf{drift} catches shifts a pass/fail can't.
  \end{itemize}
\end{columns}

% ----------------------------- EVALUATION -----------------------------
\begin{frame}{We evaluate the evaluator}{Evaluation \& rigour}
\begin{columns}[T]
  \column{0.57\textwidth}
  \begin{itemize}\setlength\itemsep{5pt}
    \item \textbf{pass$^{k}$ (k=8)}, the $\tau$-bench metric --- all 8 trials must pass; exposes inconsistency pass@1 hides.
    \item \textbf{Cohen's $\kappa$} --- judge-vs-human agreement on a gold set, chance-corrected (a constant ``pass'' scores $\approx 0$).
    \item \textbf{Drift} --- pass-rate, per-dimension, and semantic output-embedding shift across run windows.
    \item \textbf{Non-determinism addressed} --- calibration flips an order-sensitive verdict to \emph{uncertain}, not a guess.
  \end{itemize}
  \column{0.40\textwidth}
  \hspace*{0.1cm}\statcard{cindigo}{100\%}{pass@1 (single-trial)}\hfill
  \statcard{crose}{0\%}{pass$^{8}$: reliability gap}\\[7pt]
  \hspace*{0.1cm}\statcard{cteal}{0.60}{Cohen's $\kappa$ (moderate)}\hfill
  \statcard{camber}{70\%}{consistency rate}
  \par\vspace{7pt}\hspace*{0.1cm}{\scriptsize\color{muted}Test set: 100 seeded incidents, 10 root-cause categories. Full numbers in the Evaluation Report.}
\end{columns}
\end{frame}
